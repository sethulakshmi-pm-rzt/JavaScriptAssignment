const list = [1, 2, 3, 4]; const sum = (a, b) => a + b; const sub = (a, b) => a - b; const avg = (listItems) => listItems.reduce(sum, 0) / listItems.length; const mean = avg(list); const power = (a, b) => Math.abs(sub(a, b)); console.log("MEAN", mean); const variance = (listItems) => { const s = listItems.reduce((acc, currentValue) => { return sum(acc, Math.pow(power(currentValue, mean), 2)); }, 0) return Math.sqrt(s / listItems.length); } const v = variance(list); console.log("VARIANCE", v);
